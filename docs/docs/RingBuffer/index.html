  • バッファ取出: MLibRingBufferPop()

概要

リングバッファは書き込みインデックスと読み込みインデックスがバッファ内でループするバッファである。

本ライブラリは、バッファのエントリサイズとエントリ数を指定してリングバッファを作成し、バッファにデータを追加および取り出す事が出来る。 バッファへのデータ書き込み時に書込み先インデックスが読込み先インデックスに追いついた場合、最古のバッファデータは破棄しデータを書き込む。

構成

リングバッファはC標準ライブラリ上で機能します。

ハンドル初期化

説明 リングバッファのハンドルを作成し、リングバッファを初期化する。
作成したハンドルは解放するまでデータの追加や取り出しなどリングバッファの操作の為に必要となる。
定義 
#include <MLib/MLib.h>

extern MLibRet_t MLibRingBufferInit( MLibRingBuffer_t *pHandle,
                                     size_t           entrySize,
                                     size_t           entryNum,
                                     MLibErr_t        *pErr      );
戻り値 リングバッファハンドルの作成結果を返す。
MLIB_RET_SUCCESS成功
MLIB_RET_FAILURE失敗
引数 *pHandle out リングバッファハンドル
entrySize in リングバッファのエントリサイズ
entryNum in リングバッファのエントリ数
*pErr out エラー要因。不要の場合はNULLを指定する。
MLIB_ERR_NONEエラー無し
MLIB_ERR_NO_MEMORYメモリー不足
使用例 
MLibRet_t        retMLib;      /* 戻り値         */
MLibErr_t        errMLib;      /* エラー         */
MLibRingBuffer_t ringBuffer;   /* リングバッファ */

/* 初期化 */
retMLib = MLIB_RET_FAILURE;
errMLib = MLIB_ERR_NONE;

/* リングバッファ初期化 */
retMLib = MLibRingBufferInit( &ringBuffer,
                              sizeof ( char ),
                              256,
                              &errMLib         );

/* 初期化結果判定 */
if ( retMLib != MLIB_RET_SUCCESS ) {
   /* 失敗 */

   /* ここでerrMLib値からエラー要因を特定 */
}

ハンドル削除

説明 リングバッファを解放してハンドルの使用を終了する。
定義 
#include <MLib/MLib.h>

extern MLibRet_t MLibRingBufferExit( MLibRingBuffer_t *pHandle,
                                     MLibErr_t        *pErr     );
戻り値 リングバッファハンドルの削除結果を返す。
MLIB_RET_SUCCESS成功
MLIB_RET_FAILURE失敗
引数 *pHandle in リングバッファハンドル
*pErr out エラー要因。不要の場合はNULLを指定する。
MLIB_ERR_NONEエラー無し
MLIB_ERR_PARAMパラメータ不正
使用例 
MLibRingBuffer_t ringBuffer;   /* リングバッファ */

/* ここでリングバッファハンドルを作成 */

/* リングバッファ削除(戻り値未使用) */
MLibRingBufferDelete( &ringBuffer, NULL );

バッファ追加

説明 リングバッファにデータを追加する。 追加先がラップし、バッファが未だ取り出されていない場合は、上書きする。
定義 
#include <MLib/MLib.h>

extern MLibRet_t MLibRingBufferAdd( MLibRingBuffer_t *pHandle,
                                    void             *pData,
                                    MLibErr_t        *pErr     );
戻り値 リングバッファにデータを追加した結果を返す。
MLIB_RET_SUCCESS成功
MLIB_RET_FAILURE失敗
引数 *pHandle in リングバッファハンドル
*pData in データ
*pErr out エラー要因。不要の場合はNULLを指定する。
MLIB_ERR_NONEエラー無し
MLIB_ERR_PARAMパラメータ不正
使用例 
char             data;         /* データ         */
MLibRet_t        retMLib;      /* 戻り値         */
MLibErr_t        errMLib;      /* エラー         */
MLibRingBuffer_t ringBuffer;   /* リングバッファ */

/* 初期化 */
data    = 'a';
retMLib = MLIB_RET_FAILURE;
errMLib = MLIB_ERR_NONE;

/* ここでリングバッファハンドルを作成 */

/* バッファ追加 */
retMLib = MLibRingBufferPush( &ringBuffer,
                              &data,
                              &errMLib     );

/* 追加結果判定 */
if ( retMLib != MLIB_RET_SUCCESS ) {
    /* 失敗 */

    /* ここでerrMLib値からエラー要因を特定 */
}

バッファ取出

説明 リングバッファからデータを取り出す。
定義 
#include <MLib/MLib.h>

extern MLibRet_t MLibRingBufferPop( MLibRingBuffer_t *pHandle,
                                    void             *pData,
                                    MLibErr_t        *pErr     );
戻り値 リングバッファからのデータ取り出し結果を返す。
MLIB_RET_SUCCESS成功
MLIB_RET_FAILURE失敗
引数 *pHandle in リングバッファハンドル
*pData out データ
*pErr out エラー要因。不要の場合はNULLを指定する。
MLIB_ERR_NONEエラー無し
MLIB_ERR_PARAMパラメータ不正
使用例 
char             data;         /* データ         */
MLibRet_t        retMLib;      /* 戻り値         */
MLibErr_t        errMLib;      /* エラー要因     */
MLibRingBuffer_t ringBuffer;   /* リングバッファ */

/* 初期化 */
data    = 0;
retMLib = MLIB_RET_FAILURE;
errMLib = MLIB_ERR_NONE;

/* ここでリングバッファハンドルを生成 */

/* ここでリングバッファにデータを追加 */

/* データ取り出し */
retMLib = MLibRingBufferPop( &ringBuffer,
                             &data,
                             &errMLib     );

/* 取り出し結果判定 */
if ( retMLib != MLIB_RET_SUCCESS ) {
    /* 失敗 */

    /* ここでerrMLib値からエラー要因を特定 */
}
